        public function OnLoad() {
        if (isset($_POST['Request'])) {
            $request = $_POST['Request'];
            $type = $_POST['Type'];

            switch ($request) {
                case 'Search':
                    $this->SearchItem($type);
                    break;
                case 'Insert':
                    $this->InsertItem($type);
                    break;
                case 'Update':
                    $this->UpdateItem($type);
                    break;
                case 'Delete':
                    $this->DeleteItem($type);
                    break;
            }
        }
    }
